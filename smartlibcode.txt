9]|[01]?[0-9][0-9]?)$"; return ip.matches(ipPattern); }

// AsyncTask for network connection private class ConnectionTask extends AsyncTask<Void, Void, String> { private String baseUrl; private 
String ipAddress; private String errorMessage = null; 
public ConnectionTask(String baseUrl, String ipAddress) 
 	{ 	this.baseUrl = baseUrl; 	this.ipAddress = 
ipAddress; 
} 
	@Override 	protected 	String 
doInBackground(Void... params) { 
HttpURLConnection urlConnection = null; 
BufferedReader reader = null; 
StringBuilder response = new StringBuilder(); try 
{ 
String urlString = baseUrl + "/login.php"; 
URL url = new URL(urlString); Log.d(TAG, 
"Connecting to: " + urlString); 
urlConnection = (HttpURLConnection) url.openConnection(); urlConnection.setRequestMethod("GET"); urlConnection.setConnectTimeout(CONNECTION_TIMEOUT); urlConnection.setReadTimeout(READ_TIMEOUT); urlConnection.setRequestProperty("User-Agent", "SmartLibrary-Android"); 
int responseCode = urlConnection.getResponseCode(); 
	if 	(responseCode 	== 	HttpURLConnection.HTTP_OK) 
 	{ 	reader = new BufferedReader( 
new InputStreamReader(urlConnection.getInputStream())); 
	String line; while ((line = reader.readLine()) 	!=
	 	null) 
 	{ 	response.append(line); 
} return response.toString().trim(); 
} else { errorMessage = "Server returned error code: " + responseCode; return null; 
} 
} catch (SocketTimeoutException e) { Log.e(TAG, "Connection timeout", e);  errorMessage = "Connection timeout. Please check if server is running."; 
return null; 
} catch (UnknownHostException e) { Log.e(TAG, "Unknown host", 
e); errorMessage = "Invalid IP address or server not found."; return 
null; 
} catch (Exception e) { 
 	Log.e(TAG, 	"Connection 	error", 	e); 
errorMessage = "Connection error: " + e.getMessage(); 
 	return null; 	} finally 
 	{ 	if (urlConnection != null) 
 	{ 	urlConnection.disconnect 
(); 
 	} 	try 
 	{ 	if (reader != null) 
 	{ 	reader.close(); 
} 
} catch (IOException e) { 
Log.e(TAG, "Error closing reader", e); 
} 
} 
} 
 	@Override 	protected void 
onPostExecute(String result) { if (result != null) { handleConnectionResponse(result, ipAddress); 
 	} 	else 
	{ 	setLoadingState(false); 
showErrorDialog("Connection Failed", errorMessage != null ? errorMessage : "Unknown error occurred"); 
} 
}

}

private void animateCardClick(final View view, final Runnable action) { 
// Scale down animation view.animate()
	 	.scaleX(0.95f) 
.scaleY(0.95f) 
.setDuration(100) 
.withEndAction(new Runnable() { 
@Override 
public void run() 
 	{ 	// Scale back 
up view.animate() 
.scaleX(1f) 
.scaleY(1f) 
.setDuration(100) 
.start(); 
 	// 	Execute 	the 	action 	(navigation) 
new Handler().postDelayed(new Runnable() { 
@Override 
public void run() 
 	{ 	action.run();

}, 50); 
} 
}) 
.start(); 
} 
private void setupCardAnimations() { 
// Animate cards with staggered delay 
CardView[] cards = {cardAdmin, cardStaff, cardStudent}; 
for (int i = 0; i < cards.length; i++) {  final CardView card = cards[i]; card.setAlpha(0f); card.setTranslationY(30f); card.animate() 
.alpha(1f) 
.translationY(0f) 
.setDuration(500) 
.setStartDelay(100 + (i * 150)) 
.setInterpolator(new AccelerateDecelerateInterpolator()) 
.start(); 
} } 
private void setupTouchEffects() { 
// Add touch feedback to cards 
View[] clickableViews = {cardAdmin, cardStaff, cardStudent}; 
 	for 	(View 	view 	: 	clickableViews) 
 	{ 	view.setOnTouchListener(new View.OnTouchListener() { 
	@Override 	public boolean onTouch(View v, MotionEvent event) 
{ switch (event.getAction()) 
 	{ 	case MotionEvent.ACTION_DOWN: 
v.animate().scaleX(0.98f).scaleY(0.98f).setDuration(100).start(); break; 
 	case 	MotionEvent.ACTION_UP: 
case MotionEvent.ACTION_CANCEL: 
v.animate().scaleX(1f).scaleY(1f).setDuration(100).start(); break; 
} 
return false; 
} 
}); 
}
